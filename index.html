A web audio Javascript library coming soon!

Please note this is still a work in progress and it's far from being ready. Take a look at the github repository for more information.

Create sounds from wave forms


var sawtoothWave = new Pizzicato.Sound({ 
    wave: { type: 'sawtooth' }
});

sawtoothWave.play();

Load external audio files


var click = new Pizzicato.Sound('./audio/click.wav', function() {
    console.log('click loaded!');
    click.play();
});

Add effects


var delay = new Pizzicato.Effects.Delay({
    repetitions: 6,
    time: 0.4,
    mix: 0.5
});

birds.addEffect(delay);
birds.play();
